$validateCategory = Validator::make($request->all(), [

            'name' => 'required|string|max:255',
            'slug' => 'required|string|max:255|unique:categories,slug',
            'image' => 'required|image|mimes:jpg,jpeg,png,webp|max:2048',
            'description' => 'required|string',
            'status' => 'required|boolean',
        ]);

        if ($validateCategory->fails()) {
            return response()->json([
                'status' => 'false',
                'message' => 'Validation failed',
                'data' => $validateCategory->errors()
            ], 422);
        }

public function rules()
    {
        return [
            'name' => ['required', 'string', 'max:255'],
            'short_code' => ['nullable', 'string', 'max:255'],
            'add_as_sub_cat' => ['nullable'],
            'parent_id' => ['required_if:add_as_sub_cat,1'],
        ];
    }

    public function messages()
        {
            return [
                'parent_id.required_if' => 'Please select a parent category if you are adding this as a sub-category.',
            ];
        }
